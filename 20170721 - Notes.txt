ENVIRONMENT DIVISION.
INPUT-OUTPUT SECTION.
FILE-CONTROL.
    SELECT EMP-FILE-IN ASSIGN TO EMPFILE1.
    SELECT WORK-FILE   ASSIGN TO WRKFILE.
    SELECT EMPLOYEE-FILE ASSIGN TO EMPFILE.
DATA DIVISION.
FILE SECTION.
FD  EMP-FILE-IN
    RECORD CONTAINS 50 CHARACTERS.
01  EMP-FILE-RECORD	PIC X(50).
SD  WORK-FILE.
01  WORK-RECORD.
    03  W-EMP-ID	PIC X(05).
    03  FILLER		PIC X(45).
FD  EMPLOYEE-FILE
    RECORD CONTAINS 50 CHARACTERS.
    COPY EMPREC.
PROCEDURE DIVISION.
0000-MAIN-PARA.
    PERFORM 1000-INIT-PARA.
1000-INIT-PARA.
    PERFORM 1111-SORT-PARA.
1111-SORT-PARA.
    SORT WORK-FILE
	ON ASCENDING KEY W-EMP-ID
	USING EMP-FILE-IN
	GIVING EMPLOYEE-FILE.
